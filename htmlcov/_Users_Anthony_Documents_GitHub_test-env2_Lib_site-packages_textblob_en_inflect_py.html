237def pluralize(word, pos=NOUN, custom={}, classical=True): 

238 """ Returns the plural of a given word. 

239 For example: child -> children. 

240 Handles nouns and adjectives, using classical inflection by default 

241 (e.g. where "matrix" pluralizes to "matrices" instead of "matrixes"). 

242 The custom dictionary is for user-defined replacements. 

243 """ 

244 

245 if word in custom: 

246 return custom[word] 

247 

248 # Recursion of genitives. 

249 # Remove the apostrophe and any trailing -s, 

250 # form the plural of the resultant noun, and then append an apostrophe (dog's -> dogs'). 

251 if word.endswith("'") or word.endswith("'s"): 

252 owner = word.rstrip("'s") 

253 owners = pluralize(owner, pos, custom, classical) 

254 if owners.endswith("s"): 

255 return owners + "'" 

256 else: 

257 return owners + "'s" 

258 

259 # Recursion of compound words 

260 # (Postmasters General, mothers-in-law, Roman deities). 

261 words = word.replace("-", " ").split(" ") 

262 if len(words) > 1: 

263 if words[1] == "general" or words[1] == "General" and \ 

264 words[0] not in plural_categories["general-generals"]: 

265 return word.replace(words[0], pluralize(words[0], pos, custom, classical)) 

266 elif words[1] in plural_prepositions: 

267 return word.replace(words[0], pluralize(words[0], pos, custom, classical)) 

268 else: 

269 return word.replace(words[-1], pluralize(words[-1], pos, custom, classical)) 

270 

271 # Only a very few number of adjectives inflect. 

272 n = list(range(len(plural_rules))) 

273 if pos.startswith(ADJECTIVE): 

274 n = [0, 1] 

275 

276 # Apply pluralization rules. 

277 for i in n: 

278 ruleset = plural_rules[i] 

279 for rule in ruleset: 

280 suffix, inflection, category, classic = rule 

281 # A general rule, or a classic rule in classical mode. 

282 if category == None: 

283 if not classic or (classic and classical): 

284 if suffix.search(word) is not None: 

285 return suffix.sub(inflection, word) 

286 # A rule relating to a specific category of words. 

287 if category != None: 

288 if word in plural_categories[category] and (not classic or (classic and classical)): 

289 if suffix.search(word) is not None: 

290 return suffix.sub(inflection, word)

431def singularize(word, pos=NOUN, custom={}): 

432 

433 if word in list(custom.keys()): 

434 return custom[word] 

435 

436 # Recursion of compound words (e.g. mothers-in-law). 

437 if "-" in word: 

438 words = word.split("-") 

439 if len(words) > 1 and words[1] in plural_prepositions: 

440 return singularize(words[0], pos, custom)+"-"+"-".join(words[1:]) 

441 # dogs' => dog's 

442 if word.endswith("'"): 

443 return singularize(word[:-1]) + "'s" 

444 

445 lower = word.lower() 

446 for w in singular_uninflected: 

447 if w.endswith(lower): 

448 return word 

449 for w in singular_uncountable: 

450 if w.endswith(lower): 

451 return word 

452 for w in singular_ie: 

453 if lower.endswith(w+"s"): 

454 return w 

455 for w in singular_s: 

456 if lower.endswith(w + 'es'): 

457 return w 

458 for w in list(singular_irregular.keys()): 

459 if lower.endswith(w): 

460 return re.sub('(?i)'+w+'$', singular_irregular[w], word) 

461 

462 for rule in singular_rules:
