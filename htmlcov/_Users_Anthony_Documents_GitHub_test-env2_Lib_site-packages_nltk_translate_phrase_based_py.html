10def extract( 

11 f_start, 

12 f_end, 

13 e_start, 

14 e_end, 

15 alignment, 

16 f_aligned, 

17 srctext, 

18 trgtext, 

19 srclen, 

20 trglen, 

21 max_phrase_length, 

22): 

23 """ 

24 This function checks for alignment point consistency and extracts 

25 phrases using the chunk of consistent phrases. 

26 

27 A phrase pair (e, f ) is consistent with an alignment A if and only if: 

28 

29 (i) No English words in the phrase pair are aligned to words outside it. 

30 

31 ∀e i ∈ e, (e i , f j ) ∈ A ⇒ f j ∈ f 

32 

33 (ii) No Foreign words in the phrase pair are aligned to words outside it. 

34 

35 ∀f j ∈ f , (e i , f j ) ∈ A ⇒ e i ∈ e 

36 

37 (iii) The phrase pair contains at least one alignment point. 

38 

39 ∃e i ∈ e ̄ , f j ∈ f ̄ s.t. (e i , f j ) ∈ A 

40 

41 :type f_start: int 

42 :param f_start: Starting index of the possible foreign language phrases 

43 :type f_end: int 

44 :param f_end: End index of the possible foreign language phrases 

45 :type e_start: int 

46 :param e_start: Starting index of the possible source language phrases 

47 :type e_end: int 

48 :param e_end: End index of the possible source language phrases 

49 :type srctext: list 

50 :param srctext: The source language tokens, a list of string. 

51 :type trgtext: list 

52 :param trgtext: The target language tokens, a list of string. 

53 :type srclen: int 

54 :param srclen: The number of tokens in the source language tokens. 

55 :type trglen: int 

56 :param trglen: The number of tokens in the target language tokens. 

57 """ 

58 

59 if f_end < 0: # 0-based indexing. 

60 return {} 

61 # Check if alignment points are consistent. 

62 for e, f in alignment: 

63 if (f_start <= f <= f_end) and (e < e_start or e > e_end): 

64 return {} 

65 

66 # Add phrase pairs (incl. additional unaligned f) 

67 phrases = set() 

68 fs = f_start 

69 while True: 

70 fe = min(f_end, f_start + max_phrase_length - 1) 

71 while True: 

72 # add phrase pair ([e_start, e_end], [fs, fe]) to set E 

73 # Need to +1 in range to include the end-point. 

74 src_phrase = " ".join(srctext[e_start : e_end + 1]) 

75 trg_phrase = " ".join(trgtext[fs : fe + 1]) 

76 # Include more data for later ordering. 

77 phrases.add( 

78 ((e_start, e_end + 1), (fs, fe + 1), src_phrase, trg_phrase) 

79 ) 

80 fe += 1 

81 if fe in f_aligned or fe >= trglen: 

82 break 

83 fs -= 1 

84 if fs in f_aligned or fs < 0: 

85 break 

86 return phrases

89def phrase_extraction(srctext, trgtext, alignment, max_phrase_length=0): 

90 """ 

91 Phrase extraction algorithm extracts all consistent phrase pairs from 

92 a word-aligned sentence pair. 

93 

94 The idea is to loop over all possible source language (e) phrases and find 

95 the minimal foreign phrase (f) that matches each of them. Matching is done 

96 by identifying all alignment points for the source phrase and finding the 

97 shortest foreign phrase that includes all the foreign counterparts for the 

98 source words. 

99 

100 In short, a phrase alignment has to 

101 (a) contain all alignment points for all covered words 

102 (b) contain at least one alignment point 

103 

104 >>> srctext = "michael assumes that he will stay in the house" 

105 >>> trgtext = "michael geht davon aus , dass er im haus bleibt" 

106 >>> alignment = [(0,0), (1,1), (1,2), (1,3), (2,5), (3,6), (4,9), 

107 ... (5,9), (6,7), (7,7), (8,8)] 

108 >>> phrases = phrase_extraction(srctext, trgtext, alignment) 

109 >>> for i in sorted(phrases): 

110 ... print(i) 

111 ... 

112 ((0, 1), (0, 1), 'michael', 'michael') 

113 ((0, 2), (0, 4), 'michael assumes', 'michael geht davon aus') 

114 ((0, 2), (0, 5), 'michael assumes', 'michael geht davon aus ,') 

115 ((0, 3), (0, 6), 'michael assumes that', 'michael geht davon aus , dass') 

116 ((0, 4), (0, 7), 'michael assumes that he', 'michael geht davon aus , dass er') 

117 ((0, 9), (0, 10), 'michael assumes that he will stay in the house', 'michael geht davon aus , dass er im haus bleibt') 

118 ((1, 2), (1, 4), 'assumes', 'geht davon aus') 

119 ((1, 2), (1, 5), 'assumes', 'geht davon aus ,') 

120 ((1, 3), (1, 6), 'assumes that', 'geht davon aus , dass') 

121 ((1, 4), (1, 7), 'assumes that he', 'geht davon aus , dass er') 

122 ((1, 9), (1, 10), 'assumes that he will stay in the house', 'geht davon aus , dass er im haus bleibt') 

123 ((2, 3), (4, 6), 'that', ', dass') 

124 ((2, 3), (5, 6), 'that', 'dass') 

125 ((2, 4), (4, 7), 'that he', ', dass er') 

126 ((2, 4), (5, 7), 'that he', 'dass er') 

127 ((2, 9), (4, 10), 'that he will stay in the house', ', dass er im haus bleibt') 

128 ((2, 9), (5, 10), 'that he will stay in the house', 'dass er im haus bleibt') 

129 ((3, 4), (6, 7), 'he', 'er') 

130 ((3, 9), (6, 10), 'he will stay in the house', 'er im haus bleibt') 

131 ((4, 6), (9, 10), 'will stay', 'bleibt') 

132 ((4, 9), (7, 10), 'will stay in the house', 'im haus bleibt') 

133 ((6, 8), (7, 8), 'in the', 'im') 

134 ((6, 9), (7, 9), 'in the house', 'im haus') 

135 ((8, 9), (8, 9), 'house', 'haus') 

136 

137 :type srctext: str 

138 :param srctext: The sentence string from the source language. 

139 :type trgtext: str 

140 :param trgtext: The sentence string from the target language. 

141 :type alignment: list(tuple) 

142 :param alignment: The word alignment outputs as list of tuples, where 

143 the first elements of tuples are the source words' indices and 

144 second elements are the target words' indices. This is also the output 

145 format of nltk.translate.ibm1 

146 :rtype: list(tuple) 

147 :return: A list of tuples, each element in a list is a phrase and each 

148 phrase is a tuple made up of (i) its source location, (ii) its target 

149 location, (iii) the source phrase and (iii) the target phrase. The phrase 

150 list of tuples represents all the possible phrases extracted from the 

151 word alignments. 

152 :type max_phrase_length: int 

153 :param max_phrase_length: maximal phrase length, if 0 or not specified 

154 it is set to a length of the longer sentence (srctext or trgtext). 

155 """ 

156 

157 srctext = srctext.split() # e 

158 trgtext = trgtext.split() # f 

159 srclen = len(srctext) # len(e) 

160 trglen = len(trgtext) # len(f) 

161 # Keeps an index of which source/target words that are aligned. 

162 f_aligned = [j for _, j in alignment] 

163 max_phrase_length = max_phrase_length or max(srclen, trglen) 

164 

165 # set of phrase pairs BP 

166 bp = set() 

167 

168 for e_start in range(srclen): 

169 max_idx = min(srclen, e_start + max_phrase_length) 

170 for e_end in range(e_start, max_idx): 

171 # // find the minimally matching foreign phrase 

172 # (f start , f end ) = ( length(f), 0 ) 

173 # f_start ∈ [0, len(f) - 1]; f_end ∈ [0, len(f) - 1] 

174 f_start, f_end = trglen - 1, -1 # 0-based indexing 

175 

176 for e, f in alignment: 

177 if e_start <= e <= e_end: 

178 f_start = min(f, f_start) 

179 f_end = max(f, f_end) 

180 # add extract (f start , f end , e start , e end ) to set BP 

181 phrases = extract( 

182 f_start, 

183 f_end, 

184 e_start, 

185 e_end, 

186 alignment, 

187 f_aligned, 

188 srctext, 

189 trgtext, 

190 srclen, 

191 trglen, 

192 max_phrase_length, 

193 ) 

194 if phrases: 

195 bp.update(phrases) 

196 return bp
