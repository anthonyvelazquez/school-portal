164# the value of the constants correspond to Penn Treebank.

172 """ Returns a (token, tag)-tuple with a simplified universal part-of-speech tag. 

173 """

175 return (token, "%s-%s" % (NOUN, tag.split("-")[-1])) 

176 if tag in ("NN", "NNS", "NNP", "NNPS", "NP"): 

177 return (token, NOUN) 

178 if tag in ("MD", "VB", "VBD", "VBG", "VBN", "VBP", "VBZ"): 

179 return (token, VERB) 

180 if tag in ("JJ", "JJR", "JJS"): 

181 return (token, ADJ) 

182 if tag in ("RB", "RBR", "RBS", "WRB"): 

183 return (token, ADV) 

184 if tag in ("PRP", "PRP$", "WP", "WP$"): 

185 return (token, PRON) 

186 if tag in ("DT", "PDT", "WDT", "EX"): 

187 return (token, DET) 

188 if tag in ("IN",): 

189 return (token, PREP) 

190 if tag in ("CD",): 

191 return (token, NUM) 

192 if tag in ("CC",): 

193 return (token, CONJ)

256def find_tokens(string, punctuation=PUNCTUATION, abbreviations=ABBREVIATIONS, replace=replacements, linebreak=r"\n{2,}"): 

257 """ Returns a list of sentences. Each sentence is a space-separated string of tokens (words). 

258 Handles common cases of abbreviations (e.g., etc., ...). 

259 Punctuation marks are split from other words. Periods (or ?!) mark the end of a sentence. 

260 Headings without an ending period are inferred by line breaks. 

261 """ 

262 # Handle periods separately. 

263 punctuation = tuple(punctuation.replace(".", "")) 

264 # Handle replacements (contractions). 

265 for a, b in list(replace.items()): 

266 string = re.sub(a, b, string) 

267 # Handle Unicode quotes. 

268 if isinstance(string, unicode): 

269 string = unicode(string).replace("“", " “ ")\ 

270 .replace("”", " ” ")\ 

271 .replace("‘", " ‘ ")\ 

272 .replace("’", " ’ ")\ 

273 .replace("'", " ' ")\ 

274 .replace('"', ' " ') 

275 # Collapse whitespace. 

276 string = re.sub("\r\n", "\n", string) 

277 string = re.sub(linebreak, " %s " % EOS, string) 

278 string = re.sub(r"\s+", " ", string) 

279 tokens = [] 

280 for t in TOKEN.findall(string+" "): 

281 if len(t) > 0: 

282 tail = [] 

283 while t.startswith(punctuation) and \ 

284 not t in replace: 

285 # Split leading punctuation. 

286 if t.startswith(punctuation): 

287 tokens.append(t[0]); t=t[1:] 

288 while t.endswith(punctuation+(".",)) and \ 

289 not t in replace: 

290 # Split trailing punctuation. 

291 if t.endswith(punctuation): 

292 tail.append(t[-1]); t=t[:-1] 

293 # Split ellipsis (...) before splitting period. 

294 if t.endswith("..."): 

295 tail.append("..."); t=t[:-3].rstrip(".") 

296 # Split period (if not an abbreviation). 

297 if t.endswith("."): 

298 if t in abbreviations or \ 

299 RE_ABBR1.match(t) is not None or \ 

300 RE_ABBR2.match(t) is not None or \ 

301 RE_ABBR3.match(t) is not None: 

302 break 

303 else: 

304 tail.append(t[-1]); t=t[:-1] 

305 if t != "": 

306 tokens.append(t) 

307 tokens.extend(reversed(tail)) 

308 sentences, i, j = [[]], 0, 0 

309 while j < len(tokens): 

310 if tokens[j] in ("...", ".", "!", "?", EOS): 

311 # Handle citations, trailing parenthesis, repeated punctuation (!?). 

312 while j < len(tokens) \

315 break # Balanced quotes. 

316 j += 1 

317 sentences[-1].extend(t for t in tokens[i:j] if t != EOS) 

318 sentences.append([]) 

319 i = j 

320 j += 1 

321 sentences[-1].extend(tokens[i:j]) 

322 sentences = (" ".join(s) for s in sentences if len(s) > 0) 

323 sentences = (RE_SARCASM.sub("(!)", s) for s in sentences) 

324 sentences = [RE_EMOTICONS.sub( 

325 lambda m: m.group(1).replace(" ", "") + m.group(2), s) for s in sentences] 

326 return sentences

437 """ Applies lexical rules to the given token, which is a [word, tag] list. 

438 """ 

439 w = token[0] 

440 for r in self:

442 f, x, pos, cmd = bool(0), r[0], r[-2], r[1].lower()

446 continue 

447 if (cmd == "char" and x in w) \ 

448 or (cmd == "haspref" and w.startswith(x)) \

661### SENTIMENT POLARITY LEXICON ##################################################################### 

662# A sentiment lexicon can be used to discern objective facts from subjective opinions in text. 

663# Each word in the lexicon has scores for: 

664# 1) polarity: negative vs. positive (-1.0 => +1.0) 

665# 2) subjectivity: objective vs. subjective (+0.0 => +1.0) 

666# 3) intensity: modifies next word? (x0.5 => x2.0) 

667 

668# For English, adverbs are used as modifiers (e.g., "very good"). 

669# For Dutch, adverbial adjectives are used as modifiers 

670# ("hopeloos voorspelbaar", "ontzettend spannend", "verschrikkelijk goed"). 

671# Negation words (e.g., "not") reverse the polarity of the following word. 

672 

673# Sentiment()(txt) returns an averaged (polarity, subjectivity)-tuple. 

674# Sentiment().assessments(txt) returns a list of (chunk, polarity, subjectivity, label)-tuples. 

675 

676# Semantic labels are useful for fine-grained analysis, e.g., 

677# negative words + positive emoticons could indicate cynicism. 

678 

679# Semantic labels: 

680MOOD = "mood" # emoticons, emojis 

681IRONY = "irony" # sarcasm mark (!) 

682 

683NOUN, VERB, ADJECTIVE, ADVERB = \ 

684 "NN", "VB", "JJ", "RB"

732 def load(self, path=None): 

733 """ Loads the XML-file (with sentiment annotations) from the given path. 

734 By default, Sentiment.path is lazily loaded. 

735 """ 

736 # <word form="great" wordnet_id="a-01123879" pos="JJ" polarity="1.0" subjectivity="1.0" intensity="1.0" /> 

737 # <word form="damnmit" polarity="-0.75" subjectivity="1.0" label="profanity" /> 

738 if not path: 

739 path = self._path 

740 if not os.path.exists(path): 

741 return 

742 words, synsets, labels = {}, {}, {} 

743 xml = cElementTree.parse(path) 

744 xml = xml.getroot() 

745 for w in xml.findall("word"): 

746 if self._confidence is None \ 

747 or self._confidence <= float(w.attrib.get("confidence", 0.0)): 

748 w, pos, p, s, i, label, synset = ( 

749 w.attrib.get("form"), 

750 w.attrib.get("pos"),

756 ) 

757 psi = (float(p), float(s), float(i)) 

758 if w: 

759 words.setdefault(w, {}).setdefault(pos, []).append(psi) 

760 if w and label: 

761 labels[w] = label 

762 if synset: 

763 synsets.setdefault(synset, []).append(psi) 

764 self._language = xml.attrib.get("language", self._language) 

765 # Average scores of all word senses per part-of-speech tag. 

766 for w in words: 

767 words[w] = dict((pos, [avg(each) for each in zip(*psi)]) for pos, psi in words[w].items()) 

768 # Average scores of all part-of-speech tags. 

769 for w, pos in list(words.items()): 

770 words[w][None] = [avg(each) for each in zip(*pos.values())] 

771 # Average scores of all synonyms per synset. 

772 for id, psi in synsets.items(): 

773 synsets[id] = [avg(each) for each in zip(*psi)] 

774 dict.update(self, words) 

775 dict.update(self.labeler, labels) 

776 dict.update(self._synsets, synsets)

778 def synset(self, id, pos=ADJECTIVE): 

779 """ Returns a (polarity, subjectivity)-tuple for the given synset id. 

780 For example, the adjective "horrible" has id 193480 in WordNet:

785 if pos == NOUN: 

786 id = "n-" + id 

787 if pos == VERB: 

788 id = "v-" + id 

789 if pos == ADJECTIVE: 

790 id = "a-" + id 

791 if pos == ADVERB: 

792 id = "r-" + id 

793 if dict.__len__(self) == 0: 

794 self.load()

797 def __call__(self, s, negation=True, **kwargs): 

798 """ Returns a (polarity, subjectivity)-tuple for the given sentence, 

799 with polarity between -1.0 and 1.0 and subjectivity between 0.0 and 1.0. 

800 The sentence can be a string, Synset, Text, Sentence, Chunk, Word, Document, Vector. 

801 An optional weight parameter can be given, 

802 as a function that takes a list of words and returns a weight. 

803 """ 

804 def avg(assessments, weighted=lambda w: 1): 

805 s, n = 0, 0 

806 for words, score in assessments: 

807 w = weighted(words) 

808 s += w * score 

809 n += w 

810 return s / float(n or 1) 

811 # A pattern.en.wordnet.Synset. 

812 # Sentiment(synsets("horrible", "JJ")[0]) => (-0.6, 1.0) 

813 if hasattr(s, "gloss"): 

814 a = [(s.synonyms[0],) + self.synset(s.id, pos=s.pos) + (None,)] 

815 # A synset id. 

816 # Sentiment("a-00193480") => horrible => (-0.6, 1.0) (English WordNet) 

817 # Sentiment("c_267") => verschrikkelijk => (-0.9, 1.0) (Dutch Cornetto) 

818 elif isinstance(s, basestring) and RE_SYNSET.match(s) and hasattr(s, "synonyms"): 

819 a = [(s.synonyms[0],) + self.synset(s.id, pos=s.pos) + (None,)] 

820 # A string of words. 

821 # Sentiment("a horrible movie") => (-0.6, 1.0) 

822 elif isinstance(s, basestring): 

823 a = self.assessments(((w.lower(), None) for w in " ".join(self.tokenizer(s)).split()), negation) 

824 # A pattern.en.Text. 

825 elif hasattr(s, "sentences"): 

826 a = self.assessments(((w.lemma or w.string.lower(), w.pos[:2]) for w in chain(*s)), negation) 

827 # A pattern.en.Sentence or pattern.en.Chunk. 

828 elif hasattr(s, "lemmata"): 

829 a = self.assessments(((w.lemma or w.string.lower(), w.pos[:2]) for w in s.words), negation) 

830 # A pattern.en.Word. 

831 elif hasattr(s, "lemma"): 

832 a = self.assessments(((s.lemma or s.string.lower(), s.pos[:2]),), negation) 

833 # A pattern.vector.Document. 

834 # Average score = weighted average using feature weights. 

835 # Bag-of words is unordered: inject None between each two words 

836 # to stop assessments() from scanning for preceding negation & modifiers. 

837 elif hasattr(s, "terms"): 

838 a = self.assessments(chain(*(((w, None), (None, None)) for w in s)), negation) 

839 kwargs.setdefault("weight", lambda w: s.terms[w[0]]) 

840 # A dict of (word, weight)-items. 

841 elif isinstance(s, dict): 

842 a = self.assessments(chain(*(((w, None), (None, None)) for w in s)), negation) 

843 kwargs.setdefault("weight", lambda w: s[w[0]]) 

844 # A list of words. 

845 elif isinstance(s, list): 

846 a = self.assessments(((w, None) for w in s), negation) 

847 else: 

848 a = [] 

849 weight = kwargs.get("weight", lambda w: 1) # [(w, p) for w, p, s, x in a] 

850 return Score(polarity = avg( [(w, p) for w, p, s, x in a], weight ), 

851 subjectivity = avg([(w, s) for w, p, s, x in a], weight), 

852 assessments = a)

854 def assessments(self, words=[], negation=True): 

855 """ Returns a list of (chunk, polarity, subjectivity, label)-tuples for the given list of words: 

856 where chunk is a list of successive words: a known word optionally 

857 preceded by a modifier ("very good") or a negation ("not good"). 

858 """ 

859 a = [] 

860 m = None # Preceding modifier (i.e., adverb or adjective). 

861 n = None # Preceding negation (e.g., "not beautiful"). 

862 for w, pos in words: 

863 # Only assess known words, preferably by part-of-speech tag. 

864 # Including unknown words (polarity 0.0 and subjectivity 0.0) lowers the average. 

865 if w is None: 

866 continue 

867 if w in self and pos in self[w]: 

868 p, s, i = self[w][pos] 

869 # Known word not preceded by a modifier ("good"). 

870 if m is None: 

871 a.append(dict(w=[w], p=p, s=s, i=i, n=1, x=self.labeler.get(w))) 

872 # Known word preceded by a modifier ("really good"). 

873 if m is not None: 

874 a[-1]["w"].append(w) 

875 a[-1]["p"] = max(-1.0, min(p * a[-1]["i"], +1.0)) 

876 a[-1]["s"] = max(-1.0, min(s * a[-1]["i"], +1.0)) 

877 a[-1]["i"] = i 

878 a[-1]["x"] = self.labeler.get(w) 

879 # Known word preceded by a negation ("not really good"). 

880 if n is not None: 

881 a[-1]["w"].insert(0, n) 

882 a[-1]["i"] = 1.0 / a[-1]["i"] 

883 a[-1]["n"] = -1 

884 # Known word may be a negation. 

885 # Known word may be modifying the next word (i.e., it is a known adverb). 

886 m = None 

887 n = None 

888 if pos and pos in self.modifiers or any(map(self[w].__contains__, self.modifiers)): 

889 m = (w, pos) 

890 if negation and w in self.negations: 

891 n = w 

892 else: 

893 # Unknown word may be a negation ("not good"). 

894 if negation and w in self.negations: 

895 n = w 

896 # Unknown word. Retain negation across small words ("not a good"). 

897 elif n and len(w.strip("'")) > 1: 

898 n = None 

899 # Unknown word may be a negation preceded by a modifier ("really not good"). 

900 if n is not None and m is not None and (pos in self.modifiers or self.modifier(m[0])): 

901 a[-1]["w"].append(n) 

902 a[-1]["n"] = -1 

903 n = None 

904 # Unknown word. Retain modifier across small words ("really is a good"). 

905 elif m and len(w) > 2: 

906 m = None 

907 # Exclamation marks boost previous word. 

908 if w == "!" and len(a) > 0: 

909 a[-1]["w"].append("!") 

910 a[-1]["p"] = max(-1.0, min(a[-1]["p"] * 1.25, +1.0)) 

911 # Exclamation marks in parentheses indicate sarcasm. 

912 if w == "(!)": 

913 a.append(dict(w=[w], p=0.0, s=1.0, i=1.0, n=1, x=IRONY)) 

914 # EMOTICONS: {("grin", +1.0): set((":-D", ":D"))} 

915 if w.isalpha() is False and len(w) <= 5 and w not in PUNCTUATION: # speedup 

916 for (type, p), e in EMOTICONS.items(): 

917 if w in imap(lambda e: e.lower(), e): 

918 a.append(dict(w=[w], p=p, s=1.0, i=1.0, n=1, x=MOOD)) 

919 break 

920 for i in range(len(a)): 

921 w = a[i]["w"] 

922 p = a[i]["p"] 

923 s = a[i]["s"] 

924 n = a[i]["n"] 

925 x = a[i]["x"] 

926 # "not good" = slightly bad, "not bad" = slightly good. 

927 a[i] = (w, p * -0.5 if n < 0 else p, s, x) 

928 return a

944def _suffix_rules(token, tag="NN"): 

945 """ Default morphological tagging rules for English, based on word suffixes. 

946 """ 

947 if isinstance(token, (list, tuple)): 

948 token, tag = token 

949 if token.endswith("ing"): 

950 tag = "VBG" 

951 if token.endswith("ly"): 

952 tag = "RB" 

953 if token.endswith("s") and not token.endswith(("is", "ous", "ss")): 

954 tag = "NNS" 

955 if token.endswith(("able", "al", "ful", "ible", "ient", "ish", "ive", "less", "tic", "ous")) or "-" in token: 

956 tag = "JJ" 

957 if token.endswith("ed"): 

958 tag = "VBN" 

959 if token.endswith(("ate", "ify", "ise", "ize")): 

960 tag = "VBP" 

961 return [token, tag]

963def find_tags(tokens, lexicon={}, model=None, morphology=None, context=None, entities=None, default=("NN", "NNP", "CD"), language="en", map=None, **kwargs): 

964 """ Returns a list of [token, tag]-items for the given list of tokens: 

965 ["The", "cat", "purs"] => [["The", "DT"], ["cat", "NN"], ["purs", "VB"]] 

966 Words are tagged using the given lexicon of (word, tag)-items. 

967 Unknown words are tagged NN by default. 

968 Unknown words that start with a capital letter are tagged NNP (unless language="de"). 

969 Unknown words that consist only of digits and punctuation marks are tagged CD. 

970 Unknown words are then improved with morphological rules. 

971 All words are improved with contextual rules. 

972 If a model is given, uses model for unknown words instead of morphology and context. 

973 If map is a function, it is applied to each (token, tag) after applying all rules. 

974 """ 

975 tagged = [] 

976 # Tag known words. 

977 for i, token in enumerate(tokens): 

978 tagged.append([token, lexicon.get(token, i == 0 and lexicon.get(token.lower()) or None)]) 

979 # Tag unknown words. 

980 for i, (token, tag) in enumerate(tagged): 

981 prev, next = (None, None), (None, None) 

982 if i > 0: 

983 prev = tagged[i-1] 

984 if i < len(tagged) - 1: 

985 next = tagged[i+1] 

986 if tag is None or token in (model is not None and model.unknown or ()): 

987 # Use language model (i.e., SLP). 

988 if model is not None: 

989 tagged[i] = model.apply([token, None], prev, next) 

990 # Use NNP for capitalized words (except in German). 

991 elif token.istitle() and language != "de": 

992 tagged[i] = [token, default[1]] 

993 # Use CD for digits and numbers. 

994 elif CD.match(token) is not None: 

995 tagged[i] = [token, default[2]] 

996 # Use suffix rules (e.g., -ly = RB). 

997 elif morphology is not None: 

998 tagged[i] = morphology.apply([token, default[0]], prev, next) 

999 # Use suffix rules (English default). 

1000 elif language == "en": 

1001 tagged[i] = _suffix_rules([token, default[0]]) 

1002 # Use most frequent tag (NN). 

1003 else: 

1004 tagged[i] = [token, default[0]] 

1005 # Tag words by context. 

1006 if context is not None and model is None: 

1007 tagged = context.apply(tagged) 

1008 # Tag named entities. 

1009 if entities is not None: 

1010 tagged = entities.apply(tagged) 

1011 # Map tags with a custom function. 

1012 if map is not None: 

1013 tagged = [list(map(token, tag)) or [token, default[0]] for token, tag in tagged] 

1014 return tagged

1025# Chunking rules. 

1026# CHUNKS[0] = Germanic: RB + JJ precedes NN ("the round table"). 

1027# CHUNKS[1] = Romance: RB + JJ precedes or follows NN ("la table ronde", "une jolie fille"). 

1028CHUNKS = [[ 

1029 # Germanic languages: en, de, nl, ... 

1030 ( "NP", re.compile(r"(("+NN+")/)*((DT|CD|CC|CJ)/)*(("+RB+"|"+JJ+")/)*(("+NN+")/)+")), 

1031 ( "VP", re.compile(r"(((MD|"+RB+")/)*(("+VB+")/)+)+")), 

1032 ( "VP", re.compile(r"((MD)/)")), 

1033 ( "PP", re.compile(r"((IN|PP|TO)/)+")), 

1034 ("ADJP", re.compile(r"((CC|CJ|"+RB+"|"+JJ+")/)*(("+JJ+")/)+")), 

1035 ("ADVP", re.compile(r"(("+RB+"|WRB)/)+")), 

1036], [ 

1037 # Romance languages: es, fr, it, ... 

1038 ( "NP", re.compile(r"(("+NN+")/)*((DT|CD|CC|CJ)/)*(("+RB+"|"+JJ+")/)*(("+NN+")/)+(("+RB+"|"+JJ+")/)*")), 

1039 ( "VP", re.compile(r"(((MD|"+RB+")/)*(("+VB+")/)+(("+RB+")/)*)+")), 

1040 ( "VP", re.compile(r"((MD)/)")), 

1041 ( "PP", re.compile(r"((IN|PP|TO)/)+")), 

1042 ("ADJP", re.compile(r"((CC|CJ|"+RB+"|"+JJ+")/)*(("+JJ+")/)+")), 

1043 ("ADVP", re.compile(r"(("+RB+"|WRB)/)+")), 

1044]]

1046# Handle ADJP before VP, so that 

1047# RB prefers next ADJP over previous VP. 

1048CHUNKS[0].insert(1, CHUNKS[0].pop(3)) 

1049CHUNKS[1].insert(1, CHUNKS[1].pop(3))

1051def find_chunks(tagged, language="en"): 

1052 """ The input is a list of [token, tag]-items. 

1053 The output is a list of [token, tag, chunk]-items: 

1054 The/DT nice/JJ fish/NN is/VBZ dead/JJ ./. => 

1055 The/DT/B-NP nice/JJ/I-NP fish/NN/I-NP is/VBZ/B-VP dead/JJ/B-ADJP ././O 

1056 """ 

1057 chunked = [x for x in tagged] 

1058 tags = "".join("%s%s" % (tag, SEPARATOR) for token, tag in tagged) 

1059 # Use Germanic or Romance chunking rules according to given language. 

1060 for tag, rule in CHUNKS[int(language in ("ca", "es", "pt", "fr", "it", "pt", "ro"))]: 

1061 for m in rule.finditer(tags): 

1062 # Find the start of chunks inside the tags-string. 

1063 # Number of preceding separators = number of preceding tokens. 

1064 i = m.start() 

1065 j = tags[:i].count(SEPARATOR) 

1066 n = m.group(0).count(SEPARATOR) 

1067 for k in range(j, j+n): 

1068 if len(chunked[k]) == 3: 

1069 continue 

1070 if len(chunked[k]) < 3: 

1071 # A conjunction can not be start of a chunk. 

1072 if k == j and chunked[k][1] in ("CC", "CJ", "KON", "Conj(neven)"): 

1073 j += 1 

1074 # Mark first token in chunk with B-. 

1075 elif k == j: 

1076 chunked[k].append("B-"+tag) 

1077 # Mark other tokens in chunk with I-. 

1078 else: 

1079 chunked[k].append("I-"+tag) 

1080 # Mark chinks (tokens outside of a chunk) with O-. 

1081 for chink in filter(lambda x: len(x) < 3, chunked): 

1082 chink.append("O") 

1083 # Post-processing corrections. 

1084 for i, (word, tag, chunk) in enumerate(chunked): 

1085 if tag.startswith("RB") and chunk == "B-NP": 

1086 # "Very nice work" (NP) <=> "Perhaps" (ADVP) + "you" (NP). 

1087 if i < len(chunked)-1 and not chunked[i+1][1].startswith("JJ"): 

1088 chunked[i+0][2] = "B-ADVP" 

1089 chunked[i+1][2] = "B-NP" 

1090 return chunked

1092def find_prepositions(chunked): 

1093 """ The input is a list of [token, tag, chunk]-items. 

1094 The output is a list of [token, tag, chunk, preposition]-items. 

1095 PP-chunks followed by NP-chunks make up a PNP-chunk. 

1096 """ 

1097 # Tokens that are not part of a preposition just get the O-tag. 

1098 for ch in chunked: 

1099 ch.append("O") 

1100 for i, chunk in enumerate(chunked): 

1101 if chunk[2].endswith("PP") and chunk[-1] == "O": 

1102 # Find PP followed by other PP, NP with nouns and pronouns, VP with a gerund. 

1103 if i < len(chunked)-1 and \ 

1104 (chunked[i+1][2].endswith(("NP", "PP")) or \ 

1105 chunked[i+1][1] in ("VBG", "VBN")): 

1106 chunk[-1] = "B-PNP" 

1107 pp = True 

1108 for ch in chunked[i+1:]: 

1109 if not (ch[2].endswith(("NP", "PP")) or ch[1] in ("VBG", "VBN")): 

1110 break 

1111 if ch[2].endswith("PP") and pp: 

1112 ch[-1] = "I-PNP" 

1113 if not ch[2].endswith("PP"): 

1114 ch[-1] = "I-PNP" 

1115 pp = False 

1116 return chunked

1121# A shallow parser can be used to retrieve syntactic-semantic information from text 

1122# in an efficient way (usually at the expense of deeper configurational syntactic information). 

1123# The shallow parser in Pattern is meant to handle the following tasks: 

1124# 1) Tokenization: split punctuation marks from words and find sentence periods. 

1125# 2) Tagging: find the part-of-speech tag of each word (noun, verb, ...) in a sentence. 

1126# 3) Chunking: find words that belong together in a phrase. 

1127# 4) Role labeling: find the subject and object of the sentence. 

1128# 5) Lemmatization: find the base form of each word ("was" => "is"). 

1129 

1130# WORD TAG CHUNK PNP ROLE LEMMA 

1131#------------------------------------------------------------------ 

1132# The DT B-NP O NP-SBJ-1 the 

1133# black JJ I-NP O NP-SBJ-1 black 

1134# cat NN I-NP O NP-SBJ-1 cat 

1135# sat VB B-VP O VP-1 sit 

1136# on IN B-PP B-PNP PP-LOC on 

1137# the DT B-NP I-PNP NP-OBJ-1 the 

1138# mat NN I-NP I-PNP NP-OBJ-1 mat 

1139# . . O O O . 

1140 

1141# The example demonstrates what information can be retrieved: 

1142# 

1143# - the period is split from "mat." = the end of the sentence, 

1144# - the words are annotated: NN (noun), VB (verb), JJ (adjective), DT (determiner), ... 

1145# - the phrases are annotated: NP (noun phrase), VP (verb phrase), PNP (preposition), ... 

1146# - the phrases are labeled: SBJ (subject), OBJ (object), LOC (location), ... 

1147# - the phrase start is marked: B (begin), I (inside), O (outside), 

1148# - the past tense "sat" is lemmatized => "sit".

1215 def parse(self, s, tokenize=True, tags=True, chunks=True, relations=False, lemmata=False, encoding="utf-8", **kwargs): 

1216 """ Takes a string (sentences) and returns a tagged Unicode string (TaggedString). 

1217 Sentences in the output are separated by newlines. 

1218 With tokenize=True, punctuation is split from words and sentences are separated by \n. 

1219 With tags=True, part-of-speech tags are parsed (NN, VB, IN, ...). 

1220 With chunks=True, phrase chunk tags are parsed (NP, VP, PP, PNP, ...). 

1221 With relations=True, semantic role labels are parsed (SBJ, OBJ). 

1222 With lemmata=True, word lemmata are parsed. 

1223 Optional parameters are passed to 

1224 the tokenizer, tagger, chunker, labeler and lemmatizer. 

1225 """ 

1226 # Tokenizer. 

1227 if tokenize: 

1228 s = self.find_tokens(s, **kwargs) 

1229 if isinstance(s, (list, tuple)): 

1230 s = [isinstance(s, basestring) and s.split(" ") or s for s in s] 

1231 if isinstance(s, basestring): 

1232 s = [s.split(" ") for s in s.split("\n")] 

1233 # Unicode. 

1234 for i in range(len(s)): 

1235 for j in range(len(s[i])): 

1236 if isinstance(s[i][j], binary_type): 

1237 s[i][j] = decode_string(s[i][j], encoding) 

1238 # Tagger (required by chunker, labeler & lemmatizer). 

1239 if tags or chunks or relations or lemmata: 

1240 s[i] = self.find_tags(s[i], **kwargs) 

1241 else: 

1242 s[i] = [[w] for w in s[i]] 

1243 # Chunker. 

1244 if chunks or relations: 

1245 s[i] = self.find_chunks(s[i], **kwargs) 

1246 # Labeler. 

1247 if relations: 

1248 s[i] = self.find_labels(s[i], **kwargs) 

1249 # Lemmatizer. 

1250 if lemmata: 

1251 s[i] = self.find_lemmata(s[i], **kwargs) 

1252 # Slash-formatted tagged string. 

1253 # With collapse=False (or split=True), returns raw list 

1254 # (this output is not usable by tree.Text). 

1255 if not kwargs.get("collapse", True) \ 

1256 or kwargs.get("split", False): 

1257 return s 

1258 # Construct TaggedString.format. 

1259 # (this output is usable by tree.Text). 

1260 format = ["word"] 

1261 if tags: 

1262 format.append("part-of-speech") 

1263 if chunks: 

1264 format.extend(("chunk", "preposition")) 

1265 if relations: 

1266 format.append("relation") 

1267 if lemmata: 

1268 format.append("lemma") 

1269 # Collapse raw list. 

1270 # Sentences are separated by newlines, tokens by spaces, tags by slashes. 

1271 # Slashes in words are encoded with &slash; 

1272 for i in range(len(s)): 

1273 for j in range(len(s[i])): 

1274 s[i][j][0] = s[i][j][0].replace("/", "&slash;") 

1275 s[i][j] = "/".join(s[i][j]) 

1276 s[i] = " ".join(s[i]) 

1277 s = "\n".join(s) 

1278 s = TaggedString(unicode(s), format, language=kwargs.get("language", self.language)) 

1279 return s

1356 def _edit1(self, w): 

1357 """ Returns a set of words with edit distance 1 from the given word. 

1358 """ 

1359 # Of all spelling errors, 80% is covered by edit distance 1. 

1360 # Edit distance 1 = one character deleted, swapped, replaced or inserted. 

1361 split = [(w[:i], w[i:]) for i in range(len(w) + 1)] 

1362 delete, transpose, replace, insert = ( 

1363 [a + b[1:] for a, b in split if b], 

1364 [a + b[1] + b[0] + b[2:] for a, b in split if len(b) > 1], 

1365 [a + c + b[1:] for a, b in split for c in Spelling.ALPHA if b], 

1366 [a + c + b[0:] for a, b in split for c in Spelling.ALPHA] 

1367 ) 

1368 return set(delete + transpose + replace + insert)

1370 def _edit2(self, w): 

1371 """ Returns a set of words with edit distance 2 from the given word 

1372 """ 

1373 # Of all spelling errors, 99% is covered by edit distance 2. 

1374 # Only keep candidates that are actually known words (20% speedup). 

1375 return set(e2 for e1 in self._edit1(w) for e2 in self._edit1(e1) if e2 in self)

1379 """

1382 def suggest(self, w): 

1383 """ Return a list of (word, confidence) spelling corrections for the given word, 

1384 based on the probability of known words with edit distance 1-2 from the given word. 

1385 """ 

1386 if len(self) == 0: 

1387 self.load() 

1388 if len(w) == 1: 

1389 return [(w, 1.0)] # I 

1390 if w in PUNCTUATION: 

1391 return [(w, 1.0)] # .?! 

1392 if w in string.whitespace: 

1393 return [(w, 1.0)] # \n 

1394 if w.replace(".", "").isdigit(): 

1395 return [(w, 1.0)] # 1.5 

1396 candidates = self._known([w]) \ 

1397 or self._known(self._edit1(w)) \ 

1398 or self._known(self._edit2(w)) \ 

1399 or [w] 

1400 candidates = [(self.get(c, 0.0), c) for c in candidates] 

1401 s = float(sum(p for p, word in candidates) or 1) 

1402 candidates = sorted(((p / s, word) for p, word in candidates), reverse=True) 

1403 if w.istitle(): # Preserve capitalization 

1404 candidates = [(word.title(), p) for p, word in candidates] 

1405 else: 

1406 candidates = [(word, p) for p, word in candidates] 

1407 return candidates
