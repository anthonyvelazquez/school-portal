17def sentence_gleu(references, hypothesis, min_len=1, max_len=4): 

18 """ 

19 Calculates the sentence level GLEU (Google-BLEU) score described in 

20 

21 Yonghui Wu, Mike Schuster, Zhifeng Chen, Quoc V. Le, Mohammad Norouzi, 

22 Wolfgang Macherey, Maxim Krikun, Yuan Cao, Qin Gao, Klaus Macherey, 

23 Jeff Klingner, Apurva Shah, Melvin Johnson, Xiaobing Liu, Lukasz Kaiser, 

24 Stephan Gouws, Yoshikiyo Kato, Taku Kudo, Hideto Kazawa, Keith Stevens, 

25 George Kurian, Nishant Patil, Wei Wang, Cliff Young, Jason Smith, 

26 Jason Riesa, Alex Rudnick, Oriol Vinyals, Greg Corrado, Macduff Hughes, 

27 Jeffrey Dean. (2016) Google’s Neural Machine Translation System: 

28 Bridging the Gap between Human and Machine Translation. 

29 eprint arXiv:1609.08144. https://arxiv.org/pdf/1609.08144v2.pdf 

30 Retrieved on 27 Oct 2016. 

31 

32 From Wu et al. (2016): 

33 "The BLEU score has some undesirable properties when used for single 

34 sentences, as it was designed to be a corpus measure. We therefore 

35 use a slightly different score for our RL experiments which we call 

36 the 'GLEU score'. For the GLEU score, we record all sub-sequences of 

37 1, 2, 3 or 4 tokens in output and target sequence (n-grams). We then 

38 compute a recall, which is the ratio of the number of matching n-grams 

39 to the number of total n-grams in the target (ground truth) sequence, 

40 and a precision, which is the ratio of the number of matching n-grams 

41 to the number of total n-grams in the generated output sequence. Then 

42 GLEU score is simply the minimum of recall and precision. This GLEU 

43 score's range is always between 0 (no matches) and 1 (all match) and 

44 it is symmetrical when switching output and target. According to 

45 our experiments, GLEU score correlates quite well with the BLEU 

46 metric on a corpus level but does not have its drawbacks for our per 

47 sentence reward objective." 

48 

49 Note: The initial implementation only allowed a single reference, but now 

50 a list of references is required (which is consistent with 

51 bleu_score.sentence_bleu()). 

52 

53 The infamous "the the the ... " example 

54 

55 >>> ref = 'the cat is on the mat'.split() 

56 >>> hyp = 'the the the the the the the'.split() 

57 >>> sentence_gleu([ref], hyp) # doctest: +ELLIPSIS 

58 0.0909... 

59 

60 An example to evaluate normal machine translation outputs 

61 

62 >>> ref1 = str('It is a guide to action that ensures that the military ' 

63 ... 'will forever heed Party commands').split() 

64 >>> hyp1 = str('It is a guide to action which ensures that the military ' 

65 ... 'always obeys the commands of the party').split() 

66 >>> hyp2 = str('It is to insure the troops forever hearing the activity ' 

67 ... 'guidebook that party direct').split() 

68 >>> sentence_gleu([ref1], hyp1) # doctest: +ELLIPSIS 

69 0.4393... 

70 >>> sentence_gleu([ref1], hyp2) # doctest: +ELLIPSIS 

71 0.1206... 

72 

73 :param references: a list of reference sentences 

74 :type references: list(list(str)) 

75 :param hypothesis: a hypothesis sentence 

76 :type hypothesis: list(str) 

77 :param min_len: The minimum order of n-gram this function should extract. 

78 :type min_len: int 

79 :param max_len: The maximum order of n-gram this function should extract. 

80 :type max_len: int 

81 :return: the sentence level GLEU score. 

82 :rtype: float 

83 """ 

84 return corpus_gleu([references], [hypothesis], min_len=min_len, max_len=max_len)

87def corpus_gleu(list_of_references, hypotheses, min_len=1, max_len=4): 

88 """ 

89 Calculate a single corpus-level GLEU score (aka. system-level GLEU) for all 

90 the hypotheses and their respective references. 

91 

92 Instead of averaging the sentence level GLEU scores (i.e. macro-average 

93 precision), Wu et al. (2016) sum up the matching tokens and the max of 

94 hypothesis and reference tokens for each sentence, then compute using the 

95 aggregate values. 

96 

97 From Mike Schuster (via email): 

98 "For the corpus, we just add up the two statistics n_match and 

99 n_all = max(n_all_output, n_all_target) for all sentences, then 

100 calculate gleu_score = n_match / n_all, so it is not just a mean of 

101 the sentence gleu scores (in our case, longer sentences count more, 

102 which I think makes sense as they are more difficult to translate)." 

103 

104 >>> hyp1 = ['It', 'is', 'a', 'guide', 'to', 'action', 'which', 

105 ... 'ensures', 'that', 'the', 'military', 'always', 

106 ... 'obeys', 'the', 'commands', 'of', 'the', 'party'] 

107 >>> ref1a = ['It', 'is', 'a', 'guide', 'to', 'action', 'that', 

108 ... 'ensures', 'that', 'the', 'military', 'will', 'forever', 

109 ... 'heed', 'Party', 'commands'] 

110 >>> ref1b = ['It', 'is', 'the', 'guiding', 'principle', 'which', 

111 ... 'guarantees', 'the', 'military', 'forces', 'always', 

112 ... 'being', 'under', 'the', 'command', 'of', 'the', 'Party'] 

113 >>> ref1c = ['It', 'is', 'the', 'practical', 'guide', 'for', 'the', 

114 ... 'army', 'always', 'to', 'heed', 'the', 'directions', 

115 ... 'of', 'the', 'party'] 

116 

117 >>> hyp2 = ['he', 'read', 'the', 'book', 'because', 'he', 'was', 

118 ... 'interested', 'in', 'world', 'history'] 

119 >>> ref2a = ['he', 'was', 'interested', 'in', 'world', 'history', 

120 ... 'because', 'he', 'read', 'the', 'book'] 

121 

122 >>> list_of_references = [[ref1a, ref1b, ref1c], [ref2a]] 

123 >>> hypotheses = [hyp1, hyp2] 

124 >>> corpus_gleu(list_of_references, hypotheses) # doctest: +ELLIPSIS 

125 0.5673... 

126 

127 The example below show that corpus_gleu() is different from averaging 

128 sentence_gleu() for hypotheses 

129 

130 >>> score1 = sentence_gleu([ref1a], hyp1) 

131 >>> score2 = sentence_gleu([ref2a], hyp2) 

132 >>> (score1 + score2) / 2 # doctest: +ELLIPSIS 

133 0.6144... 

134 

135 :param list_of_references: a list of reference sentences, w.r.t. hypotheses 

136 :type list_of_references: list(list(list(str))) 

137 :param hypotheses: a list of hypothesis sentences 

138 :type hypotheses: list(list(str)) 

139 :param min_len: The minimum order of n-gram this function should extract. 

140 :type min_len: int 

141 :param max_len: The maximum order of n-gram this function should extract. 

142 :type max_len: int 

143 :return: The corpus-level GLEU score. 

144 :rtype: float 

145 """ 

146 # sanity check 

147 assert len(list_of_references) == len( 

148 hypotheses 

149 ), "The number of hypotheses and their reference(s) should be the same" 

150 

151 # sum matches and max-token-lengths over all sentences 

152 corpus_n_match = 0 

153 corpus_n_all = 0 

154 

155 for references, hypothesis in zip(list_of_references, hypotheses): 

156 hyp_ngrams = Counter(everygrams(hypothesis, min_len, max_len)) 

157 tpfp = sum(hyp_ngrams.values()) # True positives + False positives. 

158 

159 hyp_counts = [] 

160 for reference in references: 

161 ref_ngrams = Counter(everygrams(reference, min_len, max_len)) 

162 tpfn = sum(ref_ngrams.values()) # True positives + False negatives. 

163 

164 overlap_ngrams = ref_ngrams & hyp_ngrams 

165 tp = sum(overlap_ngrams.values()) # True positives. 

166 

167 # While GLEU is defined as the minimum of precision and 

168 # recall, we can reduce the number of division operations by one by 

169 # instead finding the maximum of the denominators for the precision 

170 # and recall formulae, since the numerators are the same: 

171 # precision = tp / tpfp 

172 # recall = tp / tpfn 

173 # gleu_score = min(precision, recall) == tp / max(tpfp, tpfn) 

174 n_all = max(tpfp, tpfn) 

175 

176 if n_all > 0: 

177 hyp_counts.append((tp, n_all)) 

178 

179 # use the reference yielding the highest score 

180 if hyp_counts: 

181 n_match, n_all = max(hyp_counts, key=lambda hc: hc[0] / hc[1]) 

182 corpus_n_match += n_match 

183 corpus_n_all += n_all 

184 

185 # corner case: empty corpus or empty references---don't divide by zero! 

186 if corpus_n_all == 0: 

187 gleu_score = 0.0 

188 else: 

189 gleu_score = corpus_n_match / corpus_n_all 

190 

191 return gleu_score
